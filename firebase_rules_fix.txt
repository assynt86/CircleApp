rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ---------- helpers ----------
    function signedIn() { return request.auth != null; }
    function uid() { return request.auth.uid; }

    function circlePath(circleId) {
      return /databases/$(database)/documents/circles/$(circleId);
    }
    function circleExists(circleId) { return exists(circlePath(circleId)); }
    function circleData(circleId) { return get(circlePath(circleId)).data; }

    function isCircleMember(circleId) {
      return signedIn()
        && circleExists(circleId)
        && (uid() in circleData(circleId).members);
    }

    function isCircleOwner(circleId) {
      return signedIn()
        && circleExists(circleId)
        && (circleData(circleId).ownerUid == uid());
    }

    function uniqueFieldPath(type, normalizedValue) {
      return /databases/$(database)/documents/unique_user_fields/$(type + "_" + normalizedValue);
    }

    // ---------- unique_user_fields ----------
    match /unique_user_fields/{fieldKey} {
      allow get: if signedIn();
      allow list: if false;

      allow create: if signedIn()
        && request.resource.data.uid == uid()
        && request.resource.data.type is string
        && request.resource.data.type in ['email', 'username', 'phone']
        && request.resource.data.normalizedValue is string
        && request.resource.data.normalizedValue.size() > 0
        && fieldKey == (request.resource.data.type + "_" + request.resource.data.normalizedValue)
        && !exists(/databases/$(database)/documents/unique_user_fields/$(fieldKey));

      allow update: if false;

      allow delete: if signedIn()
        && resource.data.uid == uid()
        && fieldKey == (resource.data.type + "_" + resource.data.normalizedValue);
    }

    // ---------- users (private) ----------
    match /users/{userId} {
      allow get, list: if signedIn() && uid() == userId;

      allow create: if signedIn()
        && uid() == userId
        && request.resource.data.uid == userId
        && request.resource.data.normalizedEmail is string
        && request.resource.data.normalizedEmail.size() > 0
        && request.resource.data.normalizedUsername is string
        && request.resource.data.normalizedUsername.size() > 0
        && request.resource.data.normalizedPhone is string
        && request.resource.data.normalizedPhone.size() > 0
        && existsAfter(uniqueFieldPath('email', request.resource.data.normalizedEmail))
        && getAfter(uniqueFieldPath('email', request.resource.data.normalizedEmail)).data.uid == userId
        && existsAfter(uniqueFieldPath('username', request.resource.data.normalizedUsername))
        && getAfter(uniqueFieldPath('username', request.resource.data.normalizedUsername)).data.uid == userId
        && existsAfter(uniqueFieldPath('phone', request.resource.data.normalizedPhone))
        && getAfter(uniqueFieldPath('phone', request.resource.data.normalizedPhone)).data.uid == userId;

      allow update: if signedIn() && (
        (
          uid() == userId
          && request.resource.data.diff(resource.data).changedKeys().hasOnly([
            'displayName', 'photoUrl', 'autoAcceptInvites', 'fcmToken'
          ])
        ) ||
        canRemoveSelfFromOtherUsersFriends(userId)
      );

      allow delete: if signedIn()
        && uid() == userId
        && (
          !('normalizedEmail' in resource.data)
          || !existsAfter(uniqueFieldPath('email', resource.data.normalizedEmail))
        )
        && (
          !('normalizedUsername' in resource.data)
          || !existsAfter(uniqueFieldPath('username', resource.data.normalizedUsername))
        )
        && (
          !('normalizedPhone' in resource.data)
          || !existsAfter(uniqueFieldPath('phone', resource.data.normalizedPhone))
        );

      function canRemoveSelfFromOtherUsersFriends(targetUserId) {
        return uid() != targetUserId
          && ('friends' in resource.data)
          && ('friends' in request.resource.data)
          && request.resource.data.diff(resource.data).changedKeys().hasOnly(['friends'])
          && (uid() in resource.data.friends)
          && !(uid() in request.resource.data.friends)
          && request.resource.data.friends.size() == resource.data.friends.size() - 1
          && resource.data.friends.hasAll(request.resource.data.friends);
      }
    }

    // ---------- user_public ----------
    match /user_public/{userId} {
      allow get, list: if signedIn();

      allow create: if signedIn()
        && uid() == userId
        && request.resource.data.uid == userId
        && existsAfter(/databases/$(database)/documents/users/$(userId))
        && getAfter(/databases/$(database)/documents/users/$(userId)).data.username == request.resource.data.username;

      allow update: if signedIn()
        && uid() == userId
        && request.resource.data.uid == resource.data.uid
        && request.resource.data.username == resource.data.username
        && request.resource.data.diff(resource.data).changedKeys().hasOnly([
          'displayName', 'photoUrl'
        ]);

      allow delete: if signedIn() && uid() == userId;
    }

    // ---------- circles ----------
    match /circles/{circleId} {

      function isMemberDoc() {
        return signedIn() && (uid() in resource.data.members);
      }

      function isOwnerDoc() {
        return signedIn() && (resource.data.ownerUid == uid());
      }

      allow get, list: if isMemberDoc();

      allow create: if signedIn()
        && request.resource.data.ownerUid == uid()
        && request.resource.data.members.size() == 1
        && request.resource.data.members[0] == uid()
        && request.resource.data.inviteCode.size() == 6;

      allow update: if signedIn() && (
        canUpdateMembers() ||
        canUpdateNameOrBackground() ||
        canOwnerUpdateCircle()
      );

      allow delete: if isOwnerDoc();

      function canUpdateMembers() {
        return request.resource.data.diff(resource.data).changedKeys().hasOnly(['members'])
          && (
            isOwnerDoc() ||
            canNonMemberJoinSelf() ||
            canMemberLeaveSelf() ||
            canMemberAddOthersNoRemovals()
          );
      }

      function canNonMemberJoinSelf() {
        return !(uid() in resource.data.members)
          && (uid() in request.resource.data.members)
          && request.resource.data.members.size() == resource.data.members.size() + 1
          && request.resource.data.members.hasAll(resource.data.members);
      }

      function canMemberLeaveSelf() {
        return (uid() in resource.data.members)
          && !(uid() in request.resource.data.members)
          && request.resource.data.members.size() == resource.data.members.size() - 1
          && resource.data.members.hasAll(request.resource.data.members);
      }

      function canMemberAddOthersNoRemovals() {
        return (uid() in resource.data.members)
          && (uid() in request.resource.data.members)
          && request.resource.data.members.size() >= resource.data.members.size()
          && request.resource.data.members.hasAll(resource.data.members);
      }

      function canUpdateNameOrBackground() {
        return isMemberDoc()
          && request.resource.data.diff(resource.data).changedKeys().hasOnly(['name', 'backgroundUrl']);
      }

      function canOwnerUpdateCircle() {
        return isOwnerDoc()
          && request.resource.data.diff(resource.data).changedKeys().hasOnly([
            'name', 'backgroundUrl', 'status', 'closeAt', 'deleteAt', 'cleanedUp', 'members'
          ]);
      }

      match /photos/{photoId} {
        allow get, list: if isCircleMember(circleId);

        // UPDATED: Now allows both .jpg and .mp4 paths depending on mediaType
        allow create: if isCircleMember(circleId)
          && request.resource.data.uploaderUid == uid()
          && (
             (request.resource.data.mediaType == 'image' && request.resource.data.storagePath == ("circles/" + circleId + "/" + photoId + ".jpg")) ||
             (request.resource.data.mediaType == 'video' && request.resource.data.storagePath == ("circles/" + circleId + "/" + photoId + ".mp4"))
          );

        allow update: if isCircleMember(circleId)
          && request.resource.data.uploaderUid == resource.data.uploaderUid
          && request.resource.data.storagePath == resource.data.storagePath;

        allow delete: if isCircleMember(circleId)
          && (isCircleOwner(circleId) || resource.data.uploaderUid == uid());
      }
    }

    // ---------- circle_codes ----------
    match /circle_codes/{inviteCode} {
      allow get: if signedIn();
      allow list: if false;

      allow create: if signedIn()
        && request.resource.data.circleId is string
        && circleExists(request.resource.data.circleId)
        && isCircleOwner(request.resource.data.circleId);

      allow update: if signedIn()
        && circleExists(resource.data.circleId)
        && isCircleMember(resource.data.circleId)
        && request.resource.data.diff(resource.data).changedKeys().hasOnly(['circleName', 'circleBackgroundUrl', 'status']);

      allow delete: if signedIn()
        && circleExists(resource.data.circleId)
        && isCircleOwner(resource.data.circleId);
    }

    // ---------- circle_invites ----------
    match /circle_invites/{inviteId} {
      allow get, list: if signedIn()
        && (uid() == resource.data.inviteeUid || uid() == resource.data.inviterUid);

      allow create: if signedIn()
        && request.resource.data.inviterUid == uid()
        && circleExists(request.resource.data.circleId)
        && isCircleMember(request.resource.data.circleId)
        && request.resource.data.status == "pending";

      allow update: if signedIn()
        && uid() == resource.data.inviteeUid
        && request.resource.data.circleId == resource.data.circleId
        && request.resource.data.inviterUid == resource.data.inviterUid
        && request.resource.data.inviteeUid == resource.data.inviteeUid
        && request.resource.data.diff(resource.data).changedKeys().hasOnly(['status'])
        && (request.resource.data.status in ["accepted", "declined"]);

      allow delete: if false;
    }

    // ---------- friend_requests ----------
    match /friend_requests/{requestId} {
      allow get, list: if signedIn()
        && (uid() == resource.data.senderUid || uid() == resource.data.receiverUid);

      allow create: if signedIn()
        && request.resource.data.senderUid == uid()
        && request.resource.data.receiverUid is string
        && request.resource.data.receiverUid != uid()
        && request.resource.data.status == "pending";

      allow update: if signedIn()
        && request.resource.data.senderUid == resource.data.senderUid
        && request.resource.data.receiverUid == resource.data.receiverUid
        && request.resource.data.diff(resource.data).changedKeys().hasOnly(['status','timestamp'])
        && (
          (uid() == resource.data.receiverUid && request.resource.data.status in ["accepted", "declined"]) ||
          (uid() == resource.data.senderUid && request.resource.data.status == "canceled")
        );

      allow delete: if signedIn()
        && (uid() == resource.data.senderUid || uid() == resource.data.receiverUid);
    }

    // ---------- notifications ----------
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;

      match /notifications/{notifId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }

    // ---------- bugs (write-only) ----------
    match /bugs/{bugId} {
      allow create: if signedIn()
        && request.resource.data.userId == uid();
      allow get, list, update, delete: if false;
    }

    // ---------- reports (write-only) ----------
    match /reports/{reportId} {
      allow create: if signedIn()
        && request.resource.data.reporterUid == uid();
      allow get, list, update, delete: if false;
    }

    // ---------- default deny ----------
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
